int minPathForDP(){
    // 边界情况
    dp[0][0] = a[0][0];
    for(int j = 1; j <n; j++) {
        dp[0][j] = dp[0][j-1] + a[0][j];
    }
    for(int i = 1; i <m; i++) {
        dp[i][0] = dp[i-1][0] + a[i][0];
    }
    
    for(int i = 1; i <m; i++) {
        for(int j = 1; j <n; j++) {
            dp[i][j] = a[i][j] + min(dp[i-1][j], dp[i][j-1]);
        }
    }
    return dp[3][3];
}
